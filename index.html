async function modifyExistingExcel() { // 1. Récupération des valeurs avec des noms de variables clairs const stgVal = document.getElementById("inputStg").value; const semVal = document.getElementById("inputSem").value; const secVal = document.getElementById("inputSec").value; // 2. Vérification si les champs sont vides if(!stgVal || !semVal || !secVal) { alert("الرجاء ملء جميع الحقول أولاً!"); return; } try { // جلب الملف مع إضافة تكتيك لمنع التخزين المؤقت (Cache البراوزر) لضمان جلب أحدث ملف const response = await fetch('template.xlsx?v=' + new Date().getTime()); // صمام الأمان: التحقق من أن الاستجابة ناجحة وليست صفحة 404 if (!response.ok) { throw new Error("لم يتم العثور على ملف template.xlsx. تأكد من رفعه في نفس مسار index.html بنفس الاسم تماماً."); } const arrayBuffer = await response.arrayBuffer(); const workbook = new ExcelJS.Workbook(); await workbook.xlsx.load(arrayBuffer); // الطريقة الآمنة لفتح أول ورقة عمل (Sheet) في الملف const worksheet = workbook.worksheets[0]; if (!worksheet) { throw new Error("الملف مفتوح ولكنه فارغ تماماً ولا يحتوي على أي ورقة عمل (Sheet)."); } // 3. إسقاط البيانات في الخلايا باستخدام Variables الصحيحة worksheet.getCell('D2').value = stgVal; worksheet.getCell('H2').value = semVal; worksheet.getCell('C3').value = secVal; // توليد وتحميل الملف النهائي const buffer = await workbook.xlsx.writeBuffer(); const blob = new Blob([buffer], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' }); const link = document.createElement('a'); link.href = URL.createObjectURL(blob); link.download = 'Document_Final.xlsx'; link.click(); } catch (error) { alert("حدث خطأ أثناء المعالجة:\n" + error.message); console.error(error); } }
